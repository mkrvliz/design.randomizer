    let history = [];
    let count = 0;
    let usedIndices = [];

    function getTheme() {
      const card = document.getElementById('card');
      card.classList.add('flash');

      setTimeout(() => {
        // Avoid recent repeats
        let available = themes.map((_, i) => i).filter(i => !usedIndices.slice(-6).includes(i));
        if (available.length === 0) { usedIndices = []; available = themes.map((_, i) => i); }

        const idx = available[Math.floor(Math.random() * available.length)];
        usedIndices.push(idx);
        const theme = themes[idx];

        count++;
        document.getElementById('countNum').textContent = count;
        document.getElementById('counter').style.display = 'block';

        // Update card
        card.innerHTML = `
          <div class="card-number">${String(count).padStart(2, '0')}</div>
          <div class="tag">${theme.tag}</div>
          <h2 class="card-title">${theme.title}</h2>
          <p class="card-context">${theme.context}</p>
        `;

        card.classList.remove('flash');

        // Update button text
        document.getElementById('btnText').textContent = 'Следующая тема';

        // History (show last 5)
        if (count > 1) {
          history.unshift(themes[usedIndices[usedIndices.length - 2]].title);
          if (history.length > 5) history.pop();

          const wrap = document.getElementById('historyWrap');
          wrap.style.display = 'flex';
          const histEl = document.getElementById('history');
          histEl.innerHTML = history.map((t, i) =>
            `<span class="history-item" style="animation-delay:${i * 0.05}s">${t}</span>`
          ).join('');
        }
      }, 150);
    }
  </script>
